
## Arithmetic operators
+, -, *, /
Example: Let a = 2;
         Let b = 5;

         console.log(a+b)
         console.log(a*b)

## Assignment Operators
 =, +=, -=, /=, *=
 Example: Let a = 5
          Let b = a + 2

## Comparison Operators
<, >, <=, >=, !+, ==, ===
Example: Let x = 5;
        console.log(x > 6)
        console.log(x <= 5)

## Logical Operators
&&, ||, !
Example: let highIncome = true;
         let goodCreditScore = true;
         let eligibleForLoan = highIncome && goodCreditScore;
         console.log(eligibleForLoan) // true

&, |, ^, ~, <<, >>, >>>
